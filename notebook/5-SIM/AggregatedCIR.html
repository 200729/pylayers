def aggcir(CIRa,alphak,tauk):
    """ aggregation of CIR from (alphak,tauk)

    Parameters
    ----------

    alphak : float
        CIR path amplitude
    tauk : float
        CIR delay values

    """
    shy = np.shape(CIRa.y)
    x = CIRa.x
    eps = (x[1]-x[0])/2
    u = map(lambda t: np.where( (x>t-eps) & (x<=t+eps))[0][0],tauk)
    ynew  = np.zeros(len(x))
    ynew[u] = alphak
    if len(shy)>1:
       CIRa.y = np.vstack((CIRa.y,ynew))
    else:
       CIRa.y = ynew[np.newaxis,:]

    return(CIRa)
