[docs]class Vect(object):
        """
        Vector class
        ------------------------------------
        """

        def __init__(self,x=0,y=0,z=0):
                self.x = x
                self.y = y
                self.z = z

        def __add__(self,v):
                w = Vect()
                w.x = self.x + v.x
                w.y = self.y + v.y
                w.z = self.z + v.z
                return(w)

        def __sub__(self,v):
                w = Vect()
                w.x = self.x - v.x
                w.y = self.y - v.y
                w.z = self.z - v.z
                return(w)

        def __repr__(self):
                return ("("+str(self.x)+',' +
                           str(self.y)+',' +
                           str(self.z)+ ")")

        
        def __mul__(self,v):
                if isinstance(v,Vect):
                        ps = self.x*v.x+self.y*v.y+self.z*v.z
                        return(ps)
                else:
                        w = Vect()
                        w.x = v*self.x
                        w.y = v*self.y
                        w.z = v*self.z
                return(w)
        
        def __div__(self,a):
                w = Vect()
                try:
                        w.x = self.x/a
                        w.y = self.y/a
                        w.z = self.z/a
                except:
                        print("error : Vect division i")
                return(w)

        def __pow__(self,v):
                w   = Vect()
                w.x = self.y*v.z-v.y*self.z
                w.y = self.z*v.x-v.z*self.x
                w.z = self.x*v.y-v.x*self.y
                return(w)

[docs]        def col(self):
                """
                Transfer a vector in a (3x1) numpy matrix
                """
                c = np.matrix(zeros([3,1]))
                c[0] = self.x
                c[1] = self.y
                c[2] = self.z
                return(c)

                """
                Transfer a vector in a (1x3) numpy matrix
                """
                l = np.matrix(zeros([1,3]))
