[docs]def shortest_path_tree(G,nroot):
    """ create the shortest path tree
    Parameters
    ----------
    G : Graph 
    nroot : node root 
    """
    SPT       = nx.Graph()
    SPT.pos   = G.pos
    for n in G.nodes():
        if n!=nroot:
            p = nx.dijkstra_path(G,nroot,n,True)
            SPT.add_path(p)
    return(SPT)
