[docs]    def take(self, net, RAT=None, LDP=None):
        """
        Parameters
        ----------
        net
        RAT
        LDP

        if RAT= None : All RAT are processed
        """
        cd = {}

        ### select RAT
        if RAT is None:
            Rat = net.RAT

        elif isinstance(RAT, str):
            Rat = [RAT]

        ### select LDP
        if LDP is None:
            Ldp = net.LDP

        elif isinstance(LDP, str):
            Ldp = [LDP]

        for ldp in Ldp:
            cd[ldp] = {}
            for rat in Rat:
                try:
                    cd[ldp][rat] = nx.get_edge_attributes(
                        net.SubNet[rat], ldp).items()
                except:
                    pass  # if the specified RAT doesn't exist in the PN

        return cd

[docs]def merge_rules(self,RAT=None,LDP=None):
    rules = {}

    for rule in self.rule:
        rules.update(rule.take(self.PN,RAT,LDP))
    return (rules)
